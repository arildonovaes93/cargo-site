const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY); exports.handler = async (event) => { try { const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: [ { price_data: { currency: 'brl', product_data: { name: 'Envio de produto CarGo', }, unit_amount: 10000, // valor em centavos (R$100,00) }, quantity: 1, }, ], mode: 'payment', success_url: 'https://seusite.netlify.app/sucesso', cancel_url: 'https://seusite.netlify.app/cancelado', }); return { statusCode: 200, body: JSON.stringify({ url: session.url }), }; } catch (err) { return { statusCode: 500, body: err.message }; } };
